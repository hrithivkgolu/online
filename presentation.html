import pandas as pd

# Sample portfolio returns and benchmark
returns = pd.Series(np.random.normal(0.001, 0.02, 252))  # Daily returns

$$ C = S_0 N(d_1) - K e^{-rT} N(d_2) $$
where
$$ d_1 = \frac{\ln(S_0/K) + (r + \sigma^2/2)T}{\sigma\sqrt{T}} $$ $$ d_2 = d_1 - \sigma\sqrt{T} $$
